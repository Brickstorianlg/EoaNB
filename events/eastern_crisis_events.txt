	id = eastern_crisis.227
	title = eastern_crisis.227.t
	desc = eastern_crisis.227.d
	picture = GFX_report_event_april_uprising
	
	is_triggered_only = yes
	
	option = {
		name = eastern_crisis.227.a
		trigger = {
			tag = BUL
		}
		add_political_power = 150
		add_stability = 0.15
		set_temp_variable = { eoanb_prestige_change = 10 }
		eoanb_change_prestige_score = yes
		hidden_effect = {
			OTO = { country_event = { id = eastern_crisis.227 } }
			every_other_country = {
				limit = {
					NOT = { tag = OTO }
				}
				news_event = { id = eastern_crisis.228 days = 1 }
			}
		}
	}
	
	option = {
		name = eastern_crisis.227.b
		trigger = {
			tag = OTO
		}
		add_political_power = -75
		add_stability = -0.05
		set_temp_variable = { eoanb_prestige_change = -5 }
		eoanb_change_prestige_score = yes
	}
}
