#Hidden Event to Calculate Majority
country_event = {
	id = eastern_crisis.415
	title = eastern_crisis.415.t
	
	fire_only_once = yes
	hidden = yes
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			if = {
				limit = {
					AND = {
						check_variable = { eastern_crisis_cc_status_quo > eastern_crisis_cc_russian_plan }
						check_variable = { eastern_crisis_cc_status_quo > eastern_crisis_cc_austrian_plan }
					}
				}
				every_country = {
					limit = {
						has_country_flag = eastern_crisis_constantinople_conference_flag
					}
					country_event = { id = eastern_crisis.416 days = 3 }
				}
				set_global_flag = eastern_crisis_cc_status_quo_flag
				OTO = { country_event = { id = eastern_crisis.419 days = 7 } }
			}
			else_if = {
				limit = {
					AND = {
						check_variable = { eastern_crisis_cc_russian_plan > eastern_crisis_cc_status_quo }
						check_variable = { eastern_crisis_cc_russian_plan > eastern_crisis_cc_austrian_plan }
					}
				}
				every_country = {
					limit = {
						has_country_flag = eastern_crisis_constantinople_conference_flag
					}
					country_event = { id = eastern_crisis.417 days = 3 }
				}
				set_global_flag = eastern_crisis_cc_russian_plan_flag
				OTO = { country_event = { id = eastern_crisis.419 days = 7 } }
			}
			else_if = {
				limit = {
					AND = {
						check_variable = { eastern_crisis_cc_austrian_plan > eastern_crisis_cc_status_quo }
						check_variable = { eastern_crisis_cc_austrian_plan > eastern_crisis_cc_russian_plan }
					}
				}
				every_country = {
					limit = {
						has_country_flag = eastern_crisis_constantinople_conference_flag
					}
					country_event = { id = eastern_crisis.418 days = 3 }
				}
				set_global_flag = eastern_crisis_cc_austrian_plan_flag
				OTO = { country_event = { id = eastern_crisis.419 days = 7 } }
			}
			else = {
				every_country = {
					limit = {
						has_country_flag = eastern_crisis_constantinople_conference_flag
					}
					country_event = { id = eastern_crisis.416 days = 3 }
				}
				set_global_flag = eastern_crisis_cc_no_majority_flag
				OTO = { country_event = { id = eastern_crisis.419 days = 7 } }
			}
		}
	}
}
