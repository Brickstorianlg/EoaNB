#Meet the Princes
country_event = {
	id = sepoy.7
	title = sepoy.7.t
	desc = sepoy.7.d
		
	is_triggered_only = yes
	
	#Bribe the Princes
	option = {
		name = sepoy.7.a
		ai_chance = { factor = 50 }
		add_political_power = -150
		add_stability = -0.35
		if = {
			limit = {
				OR = {
					has_dlc = "Together for Victory"
					has_dlc = "Waking the Tiger"
					has_dlc = "Death or Dishonor"
				}
			}
			set_autonomy = {
				target = OUD
				autonomy_state = autonomy_princely_state
			}
			set_autonomy = {
				target = GWA
				autonomy_state = autonomy_princely_state
			}
		}
		else = {
			puppet = OUD
			puppet = GWA
		}
		country_event = sepoy.8
	}

	#Persuade the Princes
	option = {
		name = sepoy.7.b
		ai_chance = { factor = 50 }
		add_political_power = -500
		if = {
			limit = {
				OR = {
					has_dlc = "Together for Victory"
					has_dlc = "Waking the Tiger"
					has_dlc = "Death or Dishonor"
				}
			}
			set_autonomy = {
				target = OUD
				autonomy_state = autonomy_confederation_member
			}
			set_autonomy = {
				target = GWA
				autonomy_state = autonomy_confederation_member
			}
		}
		else = {
			puppet = OUD
			puppet = GWA
		}
		country_event = sepoy.8
	}
}
