### kuči clan ###
defined_text = {
	name = kuciloyalty
	text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = 3
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_absolutely_loyal
	}
	text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = 2
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_high_loyal
	}
	text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = 1
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_loyal
	}
	
		text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = 0
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_neutrality
	}
	
		text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = -1
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_disloyal
	}
	
		text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = -2
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_opponent
	}
	
		text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = -3
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_revolt
	}
	#this mean the clan is in another state. MNT can get the state but in the start MNT haven´t this state
		text = {
		trigger = {
			check_variable = {
			var = montenegro_KUCI_loyalty
			value = -5
			compare = equals
			}
		}
		localization_key = CLAN_loyalty_outer
	}
}
