					}
				}
			}
		}

		visible = {
			NOT = { has_global_flag = eastern_crisis_herzegovina_uprising_flag }
			has_country_flag = AUS_international_units_in_bosnia_flag

		fire_only_once = yes
		
		cost = 100
		
		days_remove = 60

		modifier = {
			political_power_gain = -0.1
			consumer_goods_factor = 0.1
		}
		
		remove_effect = {
			country_event = { id = eastern_crisis.12 }
			if = {
				limit = {
					HRZ = {
						exists = yes
						is_puppet_of = OTO
					}
				}
				declare_war_on = {
					target = HRZ
					type = annex_everything
				}
			}
			else_if = {
				limit = {
					OTO = {
						owns_state = 104
						owns_state = 1205
						owns_state = 1207
						owns_state = 1208
						owns_state = 1272
					}
				}
				declare_war_on = {
					target = OTO
					type = annex_everything
				}
			}
			else = {
				every_other_country = {
					limit = {
						OR = {
							owns_state = 104
							owns_state = 1205
							owns_state = 1207
							owns_state = 1208
							owns_state = 1272
						}
					}
					AUS = {
						declare_war_on = {
							target = PREV
							type = annex_everything
						}
						hidden_effect = {
							PREV = { activate_mission = eastern_crisis_mission_austrian_occupation_of_bosnia }
						}
					}
				}
			}
			activate_mission = eastern_crisis_mission_austrian_occupation_of_bosnia
			hidden_effect = {
				clr_country_flag = AUS_international_units_in_bosnia_flag
				clr_country_flag = AUS_croatian_refugee_crisis_flag
			}
		}
		
		ai_will_do = {
			base = 1000
		}
